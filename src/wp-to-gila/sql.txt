
// thumbnails
insert into gila.postmeta(post_id,vartype,value)
select p.ID,'thumbnail',c.guid from wp_posts p,wp_posts c,wp_postmeta m where m.post_id=p.ID and m.meta_key='_thumbnail_id' and m.meta_value=c.ID;


update gila.postmeta a,wp_posts b,gila.post c
set a.post_id=c.id
where a.post_id=b.ID COLLATE utf8mb4_unicode_ci and b.post_name=c.title COLLATE utf8mb4_unicode_ci
